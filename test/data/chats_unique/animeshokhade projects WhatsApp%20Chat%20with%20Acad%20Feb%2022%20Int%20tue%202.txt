        ans=0

public class Solution {
    public ArrayList<Integer> solve(ArrayList<Integer> A) {
        int n = A.size();
        ArrayList<Integer> ans = new ArrayList<Integer>();

        for(int i=0; i<n; i++){
            int count = 0;
            int num = A.get(i);
            int sq_num= (int)Math.sqrt(num);
            for(int j=1; j*j <= num; j++ ){
                if((num % j) == 0){
                    count++;
                    if(num / j != j){
                        count++;
                    }
                }
            }

           

            ans.add(count );
        }

        return ans;
        
    }
}
