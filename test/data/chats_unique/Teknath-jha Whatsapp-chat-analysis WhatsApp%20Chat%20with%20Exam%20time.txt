   if (V == 0) return 0; 
  
   // Initialize result 
   ll res = INT_MAX; 
  
   // Try every coin that has smaller value than V 
   for (int i=0; i<m; i++) 
   { 
     if (coins[i] <= V) 
     { 
         int sub_res = minCoins(coins, m, V-coins[i]); 
  
         // Check for INT_MAX to avoid overflow and see if

         if (sub_res != INT_MAX && sub_res + 1 < res)
